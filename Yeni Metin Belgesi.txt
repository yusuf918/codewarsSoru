function chessKnight(cell) {  if(cell[0]="a"||cell[0]=="h"){    if(cell[1]=="1"||cell[1]=="8")      return 2;        else if(cell[1]=="2"||cell[1]=="7")      return 3;      else        return 4;      }    else if(cell[0]=="b"||cell[0]=="g"){    if(cell[1]=="1"||cell[1]=="8")      return 5;else if(cell[1]=="2"||cell[1]=="7")  return 5;    else{return 6;}          }  else{    if(cell[1]=="1"||cell[1]=="8")      return 4;    else if(cell[1]=="2"||cell[1]=="7")      return 6;    else{      return 8;    }  }}
